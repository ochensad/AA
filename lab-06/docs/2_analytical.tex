    \par –ó—Ä–µ–Ω–∏–µ -- —ç—Ç–æ –ª–æ–∫–∞–ª—å–Ω–∞—è —Å—Ç–∞—Ç–∏—Å—Ç–∏—á–µ—Å–∫–∞—è –≤–µ–ª–∏—á–∏–Ω–∞, –≤—ã—Ä–∞–∂–∞—é—â–∞—è —ç–≤—Ä–∏—Å—Ç–∏—á–µ—Å–∫–æ–µ –∂–µ–ª–∞–Ω–∏–µ –ø–æ—Å–µ—Ç–∏—Ç—å –≥–æ—Ä–æ–¥ $ùëó$ –∏–∑ –≥–æ—Ä–æ–¥–∞ $ùëñ$ [\ref{bib:4}]. –í–∏—Ä—Ç—É–∞–ª—å–Ω—ã–π —Å–ª–µ–¥, –≤ —Å–≤–æ—é –æ—á–µ—Ä–µ–¥—å, –ø—Ä–µ–¥—Å—Ç–∞–≤–ª–µ—Ç –ø–æ–¥—Ç–≤–µ—Ä–∂–¥–µ–Ω–Ω–æ–µ –æ–ø—ã—Ç–æ–º –∫–æ–ª–æ–Ω–∏–∏ –∂–µ–ª–∞–Ω–∏–µ –ø–æ—Å–µ—Ç–∏—Ç—å –≥–æ—Ä–æ–¥ $ùëó$ –∏–∑ –≥–æ—Ä–æ–¥–∞ $ùëñ$. –í–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–Ω–æ–µ –ø—Ä–∞–≤–∏–ª–æ, –æ–ø—Ä–µ–¥–µ–ª—è—é—â–µ–µ –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç—å –ø–µ—Ä–µ—Ö–æ–¥–∞ –º—É—Ä–∞–≤—å—è $ùëò$ –∏–∑ –≥–æ—Ä–æ–¥–∞ $ùëñ$ –≤ –≥–æ—Ä–æ–¥ $ùëó$ –Ω–∞ –∏—Ç–µ—Ä–∞—Ü–∏–∏ $ùë°$ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–æ —Å–æ–≥–ª–∞—Å–Ω–æ \ref{ant-prob}:
	\begin{equation}\label{ant-prob}
		P_{ij, k}\left(t\right) = 
		\begin{cases}
		\frac{
			\displaystyle \left[\tau_{i,j}\left(t\right)\right]^{\alpha} \cdot \left[\eta_{i,j}\left(t\right)\right]^{\beta} 	
		}{
			\displaystyle \sum_{l \in J_{i, k}} \left[\tau_{i,l}\left(t\right)\right]^{\alpha} \cdot \left[\eta_{i,l}\left(t\right)\right]^{\beta} 	
		} & \text{–µ—Å–ª–∏ }j \in J_{i,k}, \\
		
		0 & \text{–∏–Ω–∞—á–µ.}
		
		\end{cases}
	\end{equation}

	\par –í —Ñ–æ—Ä–º—É–ª–µ \ref{ant-prob} –ø—Ä–∏–Ω—è—Ç—ã —Å–ª–µ–¥—É—é—â–∏–µ –æ–±–æ–∑–Ω–∞—á–µ–Ω–∏—è. –°–ø–∏—Å–æ–∫ –≥–æ—Ä–æ–¥–æ–≤, –∫–æ—Ç–æ—Ä—ã–µ –µ—â–µ –Ω–µ –ø–æ—Å–µ—â–µ–Ω—ã –º—É—Ä–∞–≤—å–µ–º $k$, –∫–æ—Ç–æ—Ä—ã–π –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –≤ –≥–æ—Ä–æ–¥–µ $i$ –æ–±–æ–∑–Ω–∞—á–µ–Ω –∫–∞–∫ $J_{i,k}$, $\eta$ -- –≤–∏–¥–∏–º–æ—Å—Ç—å, –≤–µ–ª–∏—á–∏–Ω–∞, –æ–±—Ä–∞—Ç–Ω–∞—è —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—é –º–µ–∂–¥—É –≥–æ—Ä–æ–¥–∞–º–∏ $i$ –∏ $j$. –í–∏—Ä—Ç—É–∞–ª—å–Ω—ã–π —Å–ª–µ–¥ —Ñ–µ—Ä–æ–º–æ–Ω–∞ –Ω–∞ —Ä–µ–±—Ä–µ –≥—Ä–∞—Ñ–∞ $(i, j)$ –Ω–∞ –∏—Ç–µ—Ä–∞—Ü–∏–∏ $t$ –æ–±–æ–∑–Ω–∞—á–µ–Ω –∫–∞–∫ $\tau_{i, j}\left(t\right)$. $\alpha$ -- —Ä–µ–≥—É–ª–∏—Ä—É–µ–º—ã–π –ø–∞—Ä–∞–º–µ—Ç—Ä, –∑–∞–¥–∞—é—â–∏–π –≤–µ—Å —Å–ª–µ–¥–∞ —Ñ–µ—Ä–æ–º–æ–Ω–∞, $\beta$ –∑–∞–¥–∞–µ—Ç –≤–∏–¥–∏–º–æ—Å—Ç—å –ø—Ä–∏ –≤—ã–±–æ—Ä–µ –º–∞—Ä—à—Ä—É—Ç–∞.

    \par –û–¥–Ω–∞–∫–æ, –ø—Ä–∞–≤–∏–ª–æ \ref{ant-prob} –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç –ª–∏—à—å —à–∏—Ä–∏–Ω—É –∑–æ–Ω—ã –≥–æ—Ä–æ–¥–∞ $j$ [\ref{bib:5}]: –æ–±—â–∞—è –∑–æ–Ω–∞ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è –ø–æ –ø—Ä–∏–Ω—Ü–∏–ø—É "–∫–æ–ª–µ—Å–∞ —Ä—É–ª–µ—Ç–∫–∏": –∫–∞–∂–¥—ã–π –≥–æ—Ä–æ–¥ –Ω–∞ –Ω–µ–π –∏–º–µ–µ—Ç —Å–≤–æ–π —Å–µ–∫—Ç–æ—Ä —Å –ø–ª–æ—â–∞–¥—å—é, –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–æ–π –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–∏ \ref{ant-prob}. –î–ª—è –≤—ã–±–æ—Ä–∞ –≤–±—Ä–∞—Å—ã–≤–∞–µ—Ç—Å—è —Å–ª—É—á–∞–π–Ω–æ–µ —á–∏—Å–ª–æ, –∏ –ø–æ –Ω–µ–º—É –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç—Å—è —Å–µ–∫—Ç–æ—Ä, —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–∏–π –≥–æ—Ä–æ–¥—É.

	\par –ü–æ—Å–ª–µ –∑–∞–≤–µ—Ä—à–µ–Ω–∏—è –º–∞—Ä—à—Ä—É—Ç–∞ –º—É—Ä–∞–≤–µ–π $k$ –¥–æ–ª–∂–µ–Ω –æ—Å—Ç–∞–≤–∏—Ç—å –Ω–∞ —Ä–µ–±—Ä–µ  $(i, j)$ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ñ–µ—Ä–æ–º–æ–Ω–∞ —Å–æ–≥–ª–∞—Å–Ω–æ —Å–ª–µ–¥—É—é—â–µ–º—É –ø—Ä–∞–≤–∏–ª—É \ref{pheromone}:
	\begin{equation}\label{pheromone}
	\Delta\tau_{ij, k}\left(t\right) = 
	\begin{cases}
	\displaystyle	\frac{Q}{L_k\left(t\right)} & (i, j) \in T_k\left(t\right), \\
	\displaystyle   0 & \text{–∏–Ω–∞—á–µ.}
	\end{cases}
	\end{equation}
	
	\par –ó–¥–µ—Å—å $T_k\left(t\right)$ -- –º–∞—Ä—à—Ä—É—Ç, –∫–æ—Ç–æ—Ä—ã–π –±—ã–ª –ø—Ä–æ–π–¥–µ–Ω –º—É—Ä–∞–≤—å–µ–º $k$ –∑–∞ –∏—Ç–µ—Ä–∞—Ü–∏—é $t$, $L_k\left(t\right)$ -- –¥–ª–∏–Ω–∞ —ç—Ç–æ–≥–æ –º–∞—Ä—à—Ä—É—Ç–∞, –∞ $Q$ -- —Ä–µ–≥—É–ª–∏—Ä—É–µ–º—ã–π –ø–∞—Ä–∞–º–µ—Ç—Ä, –∏–º–µ—é—â–∏–π –∑–Ω–∞—á–µ–Ω–∏–µ –ø–æ—Ä—è–¥–∫–∞ –¥–ª–∏–Ω—ã –æ–ø—Ç–∏–º–∞–ª—å–Ω–æ–≥–æ –ø—É—Ç–∏ \cite{ant-ulya}.

	\par –ò—Å–ø–∞—Ä–µ–Ω–∏–µ —Ñ–µ—Ä–æ–º–æ–Ω–∞ –æ–±–µ—Å–ø–µ—á–∏–≤–∞–µ—Ç—Å—è –ø—Ä–∞–≤–∏–ª–æ–º \ref{vanish}:
	\begin{equation}\label{vanish}
	\Delta\tau_{ij}\left(t + 1\right) = \left(1 - \rho\right) \cdot \tau_{ij}\left(t\right) + \sum^{m}_{k = 1}\Delta\tau_{ij, k}\left(t\right),
	\end{equation}
	–≥–¥–µ $m$ -- –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –º—É—Ä–∞–≤—å–µ–≤ –≤ –∫–æ–ª–æ–Ω–∏–∏.


	\par –í –Ω–∞—á–∞–ª–µ –∞–ª–≥–æ—Ä–∏—Ç–º–∞ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ñ–µ—Ä–æ–º–æ–Ω–∞ –ø—Ä–∏–Ω–∏–º–∞–µ—Ç—Å—è —Ä–∞–≤–Ω—ã–º –Ω–µ–±–æ–ª—å—à–æ–º—É –ø–æ–ª–æ–∂–∏—Ç–µ–ª—å–Ω–æ–º—É —á–∏—Å–ª—É. –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ –º—É—Ä–∞–≤—å–µ–≤ –ø–æ—Å—Ç–æ—è–Ω–Ω–æ –∏ —Ä–∞–≤–Ω–æ –∫–æ–ª–∏—á–µ—Å—Ç–≤—É –≤–µ—Ä—à–∏–Ω –≥—Ä–∞—Ñ–∞.

	\par –¢–∞–∫–∂–µ –¥–ª—è —É–ª—É—á—à–µ–Ω–∏—è –≤—Ä–µ–º–µ–Ω–Ω—ã—Ö —Ö–∞—Ä–∞–∫—Ç–µ—Ä–∏—Å—Ç–∏–∫ –º—É—Ä–∞–≤—å–∏–Ω–æ–≥–æ –∞–ª–≥–æ—Ä–∏—Ç–º–∞ –≤–≤–æ–¥—è—Ç —Ç–∞–∫ –Ω–∞–∑—ã–≤–∞–µ–º—ã—Ö —ç–ª–∏—Ç–Ω—ã—Ö –º—É—Ä–∞–≤—å–µ–≤ [\ref{bib:4}]. –≠–ª–∏—Ç–Ω—ã–π –º—É—Ä–∞–≤–µ–π —É—Å–∏–ª–∏–≤–∞–µ—Ç —Ä–µ–±—Ä–∞ –Ω–∞–∏–ª—É—á—à–µ–≥–æ –º–∞—Ä—à—Ä—É—Ç–∞, –Ω–∞–π–¥–µ–Ω–Ω–æ–≥–æ —Å –Ω–∞—á–∞–ª–∞ —Ä–∞–±–æ—Ç—ã –∞–ª–≥–æ—Ä–∏—Ç–º–∞. –ö–æ–ª–∏—á–µ—Å—Ç–≤–æ —Ñ–µ—Ä–æ–º–æ–Ω–∞, –æ—Ç–∫–ª–∞–¥—ã–≤–∞–µ–º–æ–≥–æ –Ω–∞ —Ä–µ–±—Ä–∞—Ö –Ω–∞–∏–ª—É—á—à–µ–≥–æ —Ç–µ–∫—É—â–µ–≥–æ –º–∞—Ä—à—Ä—É—Ç–∞ $T^+$, –ø—Ä–∏–Ω–∏–º–∞–µ—Ç—Å—è —Ä–∞–≤–Ω—ã–º \begin{math} \frac{Q}{L^+} \end{math}, –≥–¥–µ $L^+$ -- –¥–ª–∏–Ω–∞ –º–∞—Ä—à—Ä—É—Ç–∞ $T^+$. –≠—Ç–æ—Ç —Ñ–µ—Ä–æ–º–æ–Ω –ø–æ–±—É–∂–¥–∞–µ—Ç –º—É—Ä–∞–≤—å–µ–≤ –∫ –∏—Å—Å–ª–µ–¥–æ–≤–∞–Ω–∏—é —Ä–µ—à–µ–Ω–∏–π, —Å–æ–¥–µ—Ä–∂–∞—â–∏—Ö –Ω–µ—Å–∫–æ–ª—å–∫–æ —Ä–µ–±–µ—Ä –Ω–∞–∏–ª—É—á—à–µ–≥–æ –Ω–∞ –¥–∞–Ω–Ω—ã–π –º–æ–º–µ–Ω—Ç –º–∞—Ä—à—Ä—É—Ç–∞. –ï—Å–ª–∏ –≤ –º—É—Ä–∞–≤–µ–π–Ω–∏–∫–µ –µ—Å—Ç—å $e$ —ç–ª–∏—Ç–Ω—ã—Ö –º—É—Ä–∞–≤—å–µ–≤, —Ç–æ —Ä–µ–±—Ä–∞ –º–∞—Ä—à—Ä—É—Ç–∞ $T^+$ –±—É–¥—É—Ç –ø–æ–ª—É—á–∞—Ç—å –æ–±—â–µ–µ —É—Å–∏–ª–µ–Ω–∏–µ
	\begin{equation}\label{elit}
	\Delta\tau_{e} = e \cdot \frac{Q}{L^+}.
	\end{equation}

	\section*{–í—ã–≤–æ–¥}
\par –ü—Ä–æ–≥—Ä–∞–º–º–Ω–æ–µ –æ–±–µ—Å–ø–µ—á–µ–Ω–∏–µ, —Ä–µ—à–∞—é—â–µ–µ –ø–æ—Å—Ç–∞–≤–ª–µ–Ω–Ω—É—é –∑–∞–¥–∞—á—É, –º–æ–∂–µ—Ç —Ä–∞–±–æ—Ç–∞—Ç—å —Å–ª–µ–¥—É—é—â–∏–º –æ–±—Ä–∞–∑–æ–º. –ù–∞ –≤—Ö–æ–¥ –∞–ª–≥–æ—Ä–∏—Ç–º—É –ø–æ–¥–∞–µ—Ç—Å—è –º–∞—Ç—Ä–∏—Ü–∞ —Å–º–µ–∂–Ω–æ—Å—Ç–∏ –∏ –æ–ø–∏—Å–∞–Ω–Ω—ã–µ –≤ —Ä–∞–∑–¥–µ–ª–µ —Ä–µ–≥—É–ª–∏—Ä—É–µ–º—ã–µ –ø–∞—Ä–∞–º–µ—Ç—Ä—ã. –ü—Ä–æ–≥—Ä–∞–º–º–∞ –≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –¥–ª–∏–Ω—É –∫—Ä–∞—Ç—á–∞–π—à–µ–≥–æ –ø—É—Ç–∏. 

–ú–æ–¥–µ–ª–∏—Ä—É–µ–º—ã–π –≥—Ä–∞—Ñ —è–≤–ª—è–µ—Ç—Å—è –ø–æ–ª–Ω–æ—Å—Ç—å—é —Å–≤—è–∑–Ω—ã–º, —á—Ç–æ –æ–±–µ—Å–ø–µ—á–∏–≤–∞–µ—Ç –Ω–∞–ª–∏—á–∏–µ –ø—É—Ç–∏ –º–µ–∂–¥—É –¥–≤—É–º—è –ø—Ä–æ–∏–∑–≤–æ–ª—å–Ω—ã–º–∏ –≥–æ—Ä–æ–¥–∞–º–∏.
